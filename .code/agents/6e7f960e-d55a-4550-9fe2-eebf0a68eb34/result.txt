    } else if (error.status === 401) {

    }

  }
}

title: "$TITLE"

lifecycle:
  stage: "Active"

EOF
    
    # Prepend frontmatter
    cat "/tmp/frontmatter.yaml" "$file" > "/tmp/temp.md"
